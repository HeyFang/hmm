Experiment 1: 32-bit Addition

MOV SI, 0400H
MOV DI, 0600H
MOV AX, [SI]
MOV BX, [SI+4]
ADD AX, BX
MOV [DI], AX
MOV AX, [SI+2]
MOV BX, [SI+6]
ADC AX, BX
MOV [DI+2], AX
INT 03

Experiment 3: Ascending and Descending

Ascending:
MOV CL, 05H
UP1: MOV CH, 05H

MOV SI, 0400H
MOV DI, 0401H

UP: MOV AL, [SI]
CMP AL, [DI]
JC DOWN

MOV BL, [DI]
MOV [DI], AL
MOV [SI], BL

DOWN: INC SI
INC DI
DEC CH
JNZ UP
DEC CL
JNZ UP1
HLT

Decreasing:
MOV CL, 05H
UP1: MOV CH, 05H

MOV SI, 0400H
MOV DI, 0401H

UP: MOV AL, [SI]
CMP AL, [DI]
JC DOWN

MOV BL, [DI]
MOV [DI], AL
MOV [SI], BL

DOWN: INC SI
INC DI
DEC CH
JNZ UP
DEC CL
JNZ UP1
HLT
